import Foundation
import Security

enum KeychainHelper {
    static func save(key: String, value: String) throws {
        guard let data = value.data(using: .utf8) else { return }
        let query: [String: Any] = [
            kSecClass as String: kSecClassGenericPassword,
            kSecAttrAccount as String: key,
            kSecValueData as String: data
        ]
        SecItemDelete(query as CFDictionary)
        let status = SecItemAdd(query as CFDictionary, nil)
        guard status == errSecSuccess else {
            throw KeychainError.saveFailed(status)
        }
    }

    static func read(key: String) -> String? {
        let query: [String: Any] = [
            kSecClass as String: kSecClassGenericPassword,
            kSecAttrAccount as String: key,
            kSecReturnData as String: true,
            kSecMatchLimit as String: kSecMatchLimitOne
        ]
        var result: AnyObject?
        let status = SecItemCopyMatching(query as CFDictionary, &result)
        guard status == errSecSuccess, let data = result as? Data else { return nil }
        return String(data: data, encoding: .utf8)
    }

    static func delete(key: String) throws {
        let query: [String: Any] = [
            kSecClass as String: kSecClassGenericPassword,
            kSecAttrAccount as String: key
        ]
        let status = SecItemDelete(query as CFDictionary)
        guard status == errSecSuccess || status == errSecItemNotFound else {
            throw KeychainError.deleteFailed(status)
        }
    }

    enum KeychainError: Error {
        case saveFailed(OSStatus)
        case deleteFailed(OSStatus)
    }
}
```

import Foundation
import Observation

@Observable
final class ServerConfig {
    var serverURL: String {
        didSet { UserDefaults.standard.set(serverURL, forKey: "antServerURL") }
    }

    var apiKey: String? {
        get { KeychainHelper.read(key: "antAPIKey") }
        set {
            if let newValue {
                try? KeychainHelper.save(key: "antAPIKey", value: newValue)
            } else {
                try? KeychainHelper.delete(key: "antAPIKey")
            }
        }
    }

    var isConfigured: Bool {
        !serverURL.isEmpty && apiKey != nil
    }

    init() {
        self.serverURL = UserDefaults.standard.string(forKey: "antServerURL") ?? ""
    }

    func buildURL(path: String, query: [URLQueryItem]? = nil) -> URL? {
        var components = URLComponents(string: serverURL + path)
        components?.queryItems = query
        return components?.url
    }

    func authHeaders() -> [String: String] {
        var headers = ["Content-Type": "application/json"]
        if let key = apiKey {
            headers["Authorization"] = "Bearer \(key)"
        }
        return headers
    }
}

import Foundation

final class APIClient: Sendable {
    private let config: ServerConfig
    private let session: URLSession

    init(config: ServerConfig, session: URLSession = .shared) {
        self.config = config
        self.session = session
    }

    func buildRequest(
        path: String,
        method: String,
        query: [URLQueryItem]? = nil,
        body: Data? = nil
    ) throws -> URLRequest {
        guard let url = config.buildURL(path: path, query: query) else {
            throw APIError.invalidURL
        }
        var request = URLRequest(url: url)
        request.httpMethod = method
        for (key, value) in config.authHeaders() {
            request.setValue(value, forHTTPHeaderField: key)
        }
        request.httpBody = body
        return request
    }

    func fetch<T: Decodable>(_ type: T.Type, path: String, query: [URLQueryItem]? = nil) async throws -> T {
        let request = try buildRequest(path: path, method: "GET", query: query)
        let (data, response) = try await session.data(for: request)
        try validateResponse(response)
        return try JSONDecoder.ant.decode(T.self, from: data)
    }

    func post<T: Decodable>(_ type: T.Type, path: String, body: Encodable) async throws -> T {
        let bodyData = try JSONEncoder.ant.encode(body)
        let request = try buildRequest(path: path, method: "POST", body: bodyData)
        let (data, response) = try await session.data(for: request)
        try validateResponse(response)
        return try JSONDecoder.ant.decode(T.self, from: data)
    }

    func patch<T: Decodable>(_ type: T.Type, path: String, body: Encodable) async throws -> T {
        let bodyData = try JSONEncoder.ant.encode(body)
        let request = try buildRequest(path: path, method: "PATCH", body: bodyData)
        let (data, response) = try await session.data(for: request)
        try validateResponse(response)
        return try JSONDecoder.ant.decode(T.self, from: data)
    }

    func delete(path: String) async throws {
        let request = try buildRequest(path: path, method: "DELETE")
        let (_, response) = try await session.data(for: request)
        try validateResponse(response)
    }

    private func validateResponse(_ response: URLResponse) throws {
        guard let http = response as? HTTPURLResponse else {
            throw APIError.invalidResponse
        }
        switch http.statusCode {
        case 200...299: return
        case 400...499: throw APIError.clientError(http.statusCode)
        case 500...599: throw APIError.serverError(http.statusCode)
        default: throw APIError.unexpectedStatus(http.statusCode)
        }
    }

    enum APIError: Error, LocalizedError {
        case invalidURL
        case invalidResponse
        case clientError(Int)
        case serverError(Int)
        case unexpectedStatus(Int)

        var errorDescription: String? {
            switch self {
            case .invalidURL: "Invalid server URL"
            case .invalidResponse: "Invalid response from server"
            case .clientError(let code): "Request failed (\(code))"
            case .serverError(let code): "Server error (\(code))"
            case .unexpectedStatus(let code): "Unexpected status (\(code))"
            }
        }
    }
}

extension JSONDecoder {
    static let ant: JSONDecoder = {
        let decoder = JSONDecoder()
        decoder.keyDecodingStrategy = .convertFromSnakeCase
        decoder.dateDecodingStrategy = .iso8601
        return decoder
    }()
}

extension JSONEncoder {
    static let ant: JSONEncoder = {
        let encoder = JSONEncoder()
        encoder.keyEncodingStrategy = .convertToSnakeCase
        return encoder
    }()
}
```

import Foundation
import Network
import Observation

@Observable
final class ConnectivityMonitor {
    var isConnected = true
    var connectionType: ConnectionType = .unknown

    private let monitor = NWPathMonitor()
    private let queue = DispatchQueue(label: "connectivity-monitor")

    enum ConnectionType {
        case wifi, cellular, wired, unknown
    }

    func start() {
        monitor.pathUpdateHandler = { [weak self] path in
            DispatchQueue.main.async {
                self?.isConnected = path.status == .satisfied
                if path.usesInterfaceType(.wifi) {
                    self?.connectionType = .wifi
                } else if path.usesInterfaceType(.cellular) {
                    self?.connectionType = .cellular
                } else if path.usesInterfaceType(.wiredEthernet) {
                    self?.connectionType = .wired
                } else {
                    self?.connectionType = .unknown
                }
            }
        }
        monitor.start(queue: queue)
    }

    func stop() {
        monitor.cancel()
    }
}

import Foundation
import SocketIO
import Observation

@Observable
final class SocketClient {
    var isConnected = false

    private var manager: SocketManager?
    private var socket: SocketIOClient?
    private let config: ServerConfig

    init(config: ServerConfig) {
        self.config = config
    }

    func connect() {
        guard let url = URL(string: config.serverURL) else { return }
        var socketConfig: SocketIOClientConfiguration = [
            .log(false),
            .compress,
            .forceWebsockets(true),
            .reconnects(true),
            .reconnectWait(1),
            .reconnectWaitMax(5)
        ]
        if let key = config.apiKey {
            socketConfig.insert(.extraHeaders(["Authorization": "Bearer \(key)"]))
        }
        manager = SocketManager(socketURL: url, config: socketConfig)
        socket = manager?.defaultSocket

        socket?.on(clientEvent: .connect) { [weak self] _, _ in
            DispatchQueue.main.async { self?.isConnected = true }
        }
        socket?.on(clientEvent: .disconnect) { [weak self] _, _ in
            DispatchQueue.main.async { self?.isConnected = false }
        }
        socket?.connect()
    }

    func disconnect() {
        socket?.disconnect()
        manager = nil
        socket = nil
    }

    func emit(_ event: String, _ data: [Any] = []) {
        socket?.emit(event, data)
    }

    func on(_ event: String, callback: @escaping ([Any]) -> Void) {
        socket?.on(event) { data, _ in callback(data) }
    }

    func joinSession(_ sessionId: String) {
        emit("join_session", [sessionId])
    }

    func leaveSession(_ sessionId: String) {
        emit("leave_session", [sessionId])
    }

    func sendTerminalInput(_ sessionId: String, data: String) {
        emit("terminal_input", [["sessionId": sessionId, "data": data]])
    }
}

import Foundation
import SwiftData

@MainActor
final class PendingActionQueue {
    private let container: ModelContainer

    init(container: ModelContainer) {
        self.container = container
    }

    func enqueue(endpoint: String, method: String, body: String) {
        let context = container.mainContext
        let action = PendingAction(endpoint: endpoint, method: method, body: body)
        context.insert(action)
        try? context.save()
    }

    func allPending() throws -> [PendingAction] {
        let context = container.mainContext
        let descriptor = FetchDescriptor<PendingAction>(sortBy: [SortDescriptor(\.createdAt)])
        return try context.fetch(descriptor)
    }

    func remove(_ action: PendingAction) {
        let context = container.mainContext
        context.delete(action)
        try? context.save()
    }

    func flush(using client: APIClient) async {
        guard let pending = try? allPending() else { return }
        for action in pending {
            do {
                let request = try client.buildRequest(
                    path: action.endpoint,
                    method: action.method,
                    body: action.body.data(using: .utf8)
                )
                let (_, response) = try await URLSession.shared.data(for: request)
                if let http = response as? HTTPURLResponse, (200...499).contains(http.statusCode) {
                    // Success or client error (don't retry 4xx)
                    remove(action)
                }
                // 5xx: leave in queue for next flush attempt
            } catch {
                break // Stop flushing on network error
            }
        }
    }
}

                body: body

            self.error = error.localizedDescription
        }

            messages.append(msg)
        }
        try? container.mainContext.save()
    }

        if let msg = messages.first(where: { $0.id == messageId }) {
            msg.status = "complete"
            try? container.mainContext.save()
        }
    }

        let chunk = CachedTerminalChunk(
            id: "\(sessionId)-\(chunkIndex)",
            sessionId: sessionId, chunkIndex: chunkIndex, data: data
        )
        container.mainContext.insert(chunk)
        chunks.append(chunk)
        try? container.mainContext.save()
    }

    func sendInput(sessionId: String, data: String, using client: APIClient) async {
        do {
            let body = TerminalInputBody(data: data)
            let bodyData = try JSONEncoder.ant.encode(body)
            let request = try client.buildRequest(
                path: "/api/sessions/\(sessionId)/terminal/input",
                method: "POST", body: bodyData
            )
            let (_, _) = try await URLSession.shared.data(for: request)
        } catch {
            self.error = error.localizedDescription
        }
    }

    private func cacheChunks(_ apiChunks: [APITerminalChunk], sessionId: String) {
        let context = container.mainContext
        for api in apiChunks {
            if chunks.first(where: { $0.id == api.id }) == nil {
                let cached = CachedTerminalChunk(
                    id: api.id, sessionId: sessionId,
                    chunkIndex: api.chunkIndex, data: api.data

// MARK: - Terminal Output Parser
enum TerminalOutputParser {
    static func parse(_ raw: String) -> AttributedString {
        var result = AttributedString()
        let lines = raw.components(separatedBy: "\n")
        for (i, line) in lines.enumerated() {
            var attr = AttributedString(line)
            if line.hasPrefix("$ ") || line.hasPrefix("% ") {
                attr.foregroundColor = .init(hex: "32D583")
                attr.font = .custom("JetBrainsMono-Bold", size: 13)
            } else if line.contains("[WARN]") || line.contains("warning") {
                attr.foregroundColor = .init(hex: "FFB547")
            } else if line.contains("[ERROR]") || line.contains("error") || line.contains("ERR") {
                attr.foregroundColor = .init(hex: "E85A4F")
            } else if line.contains("[INFO]") {
                attr.foregroundColor = .init(hex: "32D583")
            } else {
                attr.foregroundColor = .init(hex: "FAFAF9")
            }
            result.append(attr)
            if i < lines.count - 1 {
                result.append(AttributedString("\n"))
            }
        }
        return result
    }
}

                )

        .onDisappear { socket.leaveSession(session.id) }

        guard messageStore == nil else { return }
        let store = MessageStore(container: modelContext.container)
        messageStore = store
        store.loadMessages(for: session.id)
        socket.joinSession(session.id)
        Task {
            let client = APIClient(config: config)
            await store.fetchMessages(for: session.id, using: client)
        }
        // Listen for real-time events
        socket.on("stream_chunk") { data in
            guard let dict = data.first as? [String: Any],
                  let sid = dict["sessionId"] as? String, sid == session.id,
                  let mid = dict["messageId"] as? String,
                  let role = dict["role"] as? String,
                  let format = dict["format"] as? String,
                  let content = dict["content"] as? String else { return }
            DispatchQueue.main.async {
                store.handleStreamChunk(sessionId: sid, messageId: mid, role: role, format: format, content: content)
            }
        }
        socket.on("message_created") { data in
            Task {
                let client = APIClient(config: config)
                await store.fetchMessages(for: session.id, using: client)
            }
        }
    }
}
